name: Build APK (no wrapper)

on:
  push:
    branches: [ main ]
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout repository
      uses: actions/checkout@v3

    - name: Setup JDK 17
      uses: actions/setup-java@v3
      with:
        distribution: 'temurin'
        java-version: 17

    - name: Download and unpack Gradle 8.4
      run: |
        curl -sS https://services.gradle.org/distributions/gradle-8.4-bin.zip -o gradle-8.4-bin.zip
        unzip -q gradle-8.4-bin.zip
        ls -la

    - name: Build APK using Gradle binary
      run: |
        chmod +x gradle-8.4/bin/gradle
        ./gradle-8.4/bin/gradle -v
        ./gradle-8.4/bin/gradle assembleDebug --no-daemon
      env:
        GRADLE_USER_HOME: ${{ github.workspace }}/.gradle

    - name: Upload APK
      uses: actions/upload-artifact@v3
      with:
        name: EduReminder-APK
        path: app/build/outputs/apk/debug/app-debug.apk
